Python中对文件得导入导出

Date 周二 13 八月 2019 By 浮绿水 Category 技术. Tags pelican publishing

导入到文件

import pymongo
import pandas as pd
from pandas import Series,DataFrame

#连接数据库
client = pymongo.MongoClient('localhost',27017)
db = client['python']

# 授权

db.authenticate(name='zhaochuan',password='123456',source='admin')
info = db['zhao']
#加载数据
data = DataFrame(list(info.find()))

# 写入CSV文件

data.to_csv('NBA.csv')

导入到mongodb中

import pymongo
import pandas as pd
# 如果设置了权限,注意xxx用户权限要可以cover到后面使用到的数据库
client = pymongo.MongoClient('127.0.0.1', 27017)
# 连接所需数据库,test为数据库名
db=client['abckg']
db.authenticate(name='xuebaohua', password='123456', source='admin')
data = db['content'].find()
for i in data:
    row = pd.DataFrame([i])
    row.to_csv('abc.csv',mode='a',index=None,header=None)
